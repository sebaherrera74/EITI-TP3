143 portSTACK_TYPE *pxPortInitialiseStack( portSTACK_TYPE *pxTopOfStack, pdTASK_CODE pxCode, void *pvParameters )
144 {
145 /* Should actually keep this struct on the stack. */
146 xParams *pxThisThreadParams = pvPortMalloc( sizeof( xParams ) );
147 
148  (void)pthread_once( &hSigSetupThread, prvSetupSignalsAndSchedulerPolicy );
149 
150  if ( (pthread_t)NULL == hMainThread )
151  {
152  hMainThread = pthread_self();
153  }
154 
155  /* No need to join the threads. */
156  pthread_attr_init( &xThreadAttributes );
157  pthread_attr_setdetachstate( &xThreadAttributes, PTHREAD_CREATE_DETACHED );
158 
159  /* Add the task parameters. */
160  pxThisThreadParams->pxCode = pxCode;
161  pxThisThreadParams->pvParams = pvParameters;
162 
163  vPortEnterCritical();
164 
165  lIndexOfLastAddedTask = prvGetFreeThreadState();
166 
167  /* Create the new pThread. */
168  if ( 0 == pthread_mutex_lock( &xSingleThreadMutex ) )
169  {
170  xSentinel = 0;
171  if ( 0 != pthread_create( &( pxThreads[ lIndexOfLastAddedTask ].hThread ), &xThreadAttributes, prvWaitForStart, (void *)pxThisThreadParams ) )
172  {
173  /* Thread create failed, signal the failure */
174  pxTopOfStack = 0;
175  }
176 
177  /* Wait until the task suspends. */
178  (void)pthread_mutex_unlock( &xSingleThreadMutex );
179  while ( xSentinel == 0 );
180  vPortExitCritical();
181  }
182 
183  return pxTopOfStack;
184 }

203 portBASE_TYPE xPortStartScheduler( void )
204 {
205 portBASE_TYPE xResult;
206 int iSignal;
207 sigset_t xSignals;
208 sigset_t xSignalToBlock;
209 sigset_t xSignalsBlocked;
210 portLONG lIndex;
211 
212  /* Establish the signals to block before they are needed. */
213  sigfillset( &xSignalToBlock );
214 
215  /* Block until the end */
216  (void)pthread_sigmask( SIG_SETMASK, &xSignalToBlock, &xSignalsBlocked );
217 
218  for ( lIndex = 0; lIndex < MAX_NUMBER_OF_TASKS; lIndex++ )
219  {
220  pxThreads[ lIndex ].uxCriticalNesting = 0;
221  }
222 
223  /* Start the timer that generates the tick ISR. Interrupts are disabled
224  here already. */
225  prvSetupTimerInterrupt();
226 
227  /* Start the first task. Will not return unless all threads are killed. */
228  vPortStartFirstTask();
229 
230  /* This is the end signal we are looking for. */
231  sigemptyset( &xSignals );
232  sigaddset( &xSignals, SIG_RESUME );
233 
234  while ( pdTRUE != xSchedulerEnd )
235  {
236  if ( 0 != sigwait( &xSignals, &iSignal ) )
237  {
238  printf( "Main thread spurious signal: %d\n", iSignal );
239  }
240  }
241 
242  printf( "Cleaning Up, Exiting.\n" );
243  /* Cleanup the mutexes */
244  xResult = pthread_mutex_destroy( &xSuspendResumeThreadMutex );
245  xResult = pthread_mutex_destroy( &xSingleThreadMutex );
246  vPortFree( (void *)pxThreads );
247 
248  /* Should not get here! */
249  return xResult;
250 }

253 void vPortEndScheduler( void )
254 {
255 portBASE_TYPE xNumberOfThreads;
256 portBASE_TYPE xResult;
257  for ( xNumberOfThreads = 0; xNumberOfThreads < MAX_NUMBER_OF_TASKS; xNumberOfThreads++ )
258  {
259  if ( ( pthread_t )NULL != pxThreads[ xNumberOfThreads ].hThread )
260  {
261  /* Kill all of the threads, they are in the detached state. */
262  xResult = pthread_cancel( pxThreads[ xNumberOfThreads ].hThread );
263  if (xResult)
264  printf("pthread_cancel error!\n");
265  }
266  }
267 
268  /* Signal the scheduler to exit its loop. */
269  xSchedulerEnd = pdTRUE;
270  (void)pthread_kill( hMainThread, SIG_RESUME );
271 }

291 void vPortExitCritical( void )
292 {
293  /* Check for unmatched exits. */
294  if ( uxCriticalNesting > 0 )
295  {
296  uxCriticalNesting--;
297  }
298 
299  /* If we have reached 0 then re-enable the interrupts. */
300  if( uxCriticalNesting == 0 )
301  {
302  /* Have we missed ticks? This is the equivalent of pending an interrupt. */
303  if ( pdTRUE == xPendYield )
304  {
305  xPendYield = pdFALSE;
306  vPortYield();
307  }
308  vPortEnableInterrupts();
309  }

313 void vPortYield( void )
314 {
315 pthread_t xTaskToSuspend;
316 pthread_t xTaskToResume;
317 
318  if ( 0 == pthread_mutex_lock( &xSingleThreadMutex ) )
319  {
320  xTaskToSuspend = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
321 
322  vTaskSwitchContext();
323 
324  xTaskToResume = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
325  if ( xTaskToSuspend != xTaskToResume )
326  {
327  /* Remember and switch the critical nesting. */
328  prvSetTaskCriticalNesting( xTaskToSuspend, uxCriticalNesting );
329  uxCriticalNesting = prvGetTaskCriticalNesting( xTaskToResume );
330  /* Switch tasks. */
331  prvResumeThread( xTaskToResume );
332  prvSuspendThread( xTaskToSuspend );
333  }
334  else
335  {
336  /* Yielding to self */
337  (void)pthread_mutex_unlock( &xSingleThreadMutex );
338  }
339  }
340 }

374 {
375 struct itimerval itimer, oitimer;
376 portTickType xMicroSeconds = portTICK_PERIOD_MS * 1000;
377 
378  /* Initialise the structure with the current timer information. */
379  if ( 0 == getitimer( TIMER_TYPE, &itimer ) )
380  {
381  /* Set the interval between timer events. */
382  itimer.it_interval.tv_sec = 0;
383  itimer.it_interval.tv_usec = xMicroSeconds;
384 
385  /* Set the current count-down. */
386  itimer.it_value.tv_sec = 0;
387  itimer.it_value.tv_usec = xMicroSeconds;
388 
389  /* Set-up the timer interrupt. */
390  if ( 0 != setitimer( TIMER_TYPE, &itimer, &oitimer ) )
391  {
392  printf( "Set Timer problem.\n" );
393  }
394  }
395  else
396  {
397  printf( "Get Timer problem.\n" );

402 void vPortSystemTickHandler( int sig )
403 {
404 pthread_t xTaskToSuspend;
405 pthread_t xTaskToResume;
406 
407  (void)(sig);
408  if ( ( pdTRUE == xInterruptsEnabled ) && ( pdTRUE != xServicingTick ) )
409  {
410  if ( 0 == pthread_mutex_trylock( &xSingleThreadMutex ) )
411  {
412  xServicingTick = pdTRUE;
413 
414  xTaskToSuspend = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
415  /* Tick Increment. */
416  xTaskIncrementTick();
417 
418  /* Select Next Task. */
419 #if ( configUSE_PREEMPTION == 1 )
420  vTaskSwitchContext();
421 #endif
422  xTaskToResume = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
423 
424  /* The only thread that can process this tick is the running thread. */
425  if ( xTaskToSuspend != xTaskToResume )
426  {
427  /* Remember and switch the critical nesting. */
428  prvSetTaskCriticalNesting( xTaskToSuspend, uxCriticalNesting );
429  uxCriticalNesting = prvGetTaskCriticalNesting( xTaskToResume );
430  /* Resume next task. */
431  prvResumeThread( xTaskToResume );
432  /* Suspend the current task. */
433  prvSuspendThread( xTaskToSuspend );
434  }
435  else
436  {
437  /* Release the lock as we are Resuming. */
438  (void)pthread_mutex_unlock( &xSingleThreadMutex );
439  }
440  xServicingTick = pdFALSE;
441  }
442  else
443  {
444  xPendYield = pdTRUE;
445  }
446  }
447  else
448  {
449  xPendYield = pdTRUE;
450  }
451 }

454 void vPortForciblyEndThread( void *pxTaskToDelete )
455 {
456 xTaskHandle hTaskToDelete = ( xTaskHandle )pxTaskToDelete;
457 pthread_t xTaskToDelete;
458 pthread_t xTaskToResume;
459 portBASE_TYPE xResult;
460 
461  if ( 0 == pthread_mutex_lock( &xSingleThreadMutex ) )
462  {
463  xTaskToDelete = prvGetThreadHandle( hTaskToDelete );
464  xTaskToResume = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
465 
466  if ( xTaskToResume == xTaskToDelete )
467  {
468  /* This is a suicidal thread, need to select a different task to run. */
469  vTaskSwitchContext();
470  xTaskToResume = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
471  }
472 
473  if ( pthread_self() != xTaskToDelete )
474  {
475  /* Cancelling a thread that is not me. */
476  if ( xTaskToDelete != ( pthread_t )NULL )
477  {
478  /* Send a signal to wake the task so that it definitely cancels. */
479  pthread_testcancel();
480  xResult = pthread_cancel( xTaskToDelete );
481  if (xResult)
482  printf("pthread_cancel error!\n");
483  /* Pthread Clean-up function will note the cancellation. */
484  }
485  (void)pthread_mutex_unlock( &xSingleThreadMutex );
486  }
487  else
488  {
489  /* Resume the other thread. */
490  prvResumeThread( xTaskToResume );
491  /* Pthread Clean-up function will note the cancellation. */
492  /* Release the execution. */
493  uxCriticalNesting = 0;
494  vPortEnableInterrupts();
495  (void)pthread_mutex_unlock( &xSingleThreadMutex );
496  /* Commit suicide */
497  pthread_exit( (void *)1 );
498  }
499  }
500 }

524 void prvSuspendSignalHandler(int sig)
525 {
526 sigset_t xSignals;
527 
528  /* Only interested in the resume signal. */
529  sigemptyset( &xSignals );
530  sigaddset( &xSignals, SIG_RESUME );
531  xSentinel = 1;
532 
533  /* Unlock the Single thread mutex to allow the resumed task to continue. */
534  if ( 0 != pthread_mutex_unlock( &xSingleThreadMutex ) )
535  {
536  printf( "Releasing someone else's lock.\n" );
537  }
538 
539  /* Wait on the resume signal. */
540  if ( 0 != sigwait( &xSignals, &sig ) )
541  {
542  printf( "SSH: Sw %d\n", sig );
543  }
544 
545  /* Will resume here when the SIG_RESUME signal is received. */
546  /* Need to set the interrupts based on the task's critical nesting. */
547  if ( uxCriticalNesting == 0 )
548  {
549  vPortEnableInterrupts();
550  }
551  else
552  {
553  vPortDisableInterrupts();
554  }
555 }

558 void prvSuspendThread( pthread_t xThreadId )
559 {
560 portBASE_TYPE xResult = pthread_mutex_lock( &xSuspendResumeThreadMutex );
561  if ( 0 == xResult )
562  {
563  /* Set-up for the Suspend Signal handler? */
564  xSentinel = 0;
565  xResult = pthread_mutex_unlock( &xSuspendResumeThreadMutex );
566  xResult = pthread_kill( xThreadId, SIG_SUSPEND );
567  if (xResult)
568  printf("pthread_kill error!\n");
569  while ( ( xSentinel == 0 ) && ( pdTRUE != xServicingTick ) )
570  {
571  sched_yield();
572  }
573  }
574 }

606 void prvSetupSignalsAndSchedulerPolicy( void )
607 {
608 /* The following code would allow for configuring the scheduling of this task as a Real-time task.
609  * The process would then need to be run with higher privileges for it to take affect.
610 int iPolicy;
611 int iResult;
612 int iSchedulerPriority;
613  iResult = pthread_getschedparam( pthread_self(), &iPolicy, &iSchedulerPriority );
614  iResult = pthread_attr_setschedpolicy( &xThreadAttributes, SCHED_FIFO );
615  iPolicy = SCHED_FIFO;
616  iResult = pthread_setschedparam( pthread_self(), iPolicy, &iSchedulerPriority ); */
617 
618 struct sigaction sigsuspendself, sigresume, sigtick;
619 portLONG lIndex;
620 
621  pxThreads = ( xThreadState *)pvPortMalloc( sizeof( xThreadState ) * MAX_NUMBER_OF_TASKS );
622  for ( lIndex = 0; lIndex < MAX_NUMBER_OF_TASKS; lIndex++ )
623  {
624  pxThreads[ lIndex ].hThread = ( pthread_t )NULL;
625  pxThreads[ lIndex ].hTask = ( xTaskHandle )NULL;
626  pxThreads[ lIndex ].uxCriticalNesting = 0;
627  }
628 
629  sigsuspendself.sa_flags = 0;
630  sigsuspendself.sa_handler = prvSuspendSignalHandler;
631  sigfillset( &sigsuspendself.sa_mask );
632 
633  sigresume.sa_flags = 0;
634  sigresume.sa_handler = prvResumeSignalHandler;
635  sigfillset( &sigresume.sa_mask );
636 
637  sigtick.sa_flags = 0;
638  sigtick.sa_handler = vPortSystemTickHandler;
639  sigfillset( &sigtick.sa_mask );
640 
641  if ( 0 != sigaction( SIG_SUSPEND, &sigsuspendself, NULL ) )
642  {
643  printf( "Problem installing SIG_SUSPEND_SELF\n" );
644  }
645  if ( 0 != sigaction( SIG_RESUME, &sigresume, NULL ) )
646  {
647  printf( "Problem installing SIG_RESUME\n" );
648  }
649  if ( 0 != sigaction( SIG_TICK, &sigtick, NULL ) )
650  {
651  printf( "Problem installing SIG_TICK\n" );
652  }
653  printf( "Running as PID: %d\n", getpid() );
654 }
