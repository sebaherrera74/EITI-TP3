99 /* The lowest interrupt priority that can be used in a call to a "set priority"
100  * function. */
101 #define configLIBRARY_LOWEST_INTERRUPT_PRIORITY 0x7

103 /* The highest interrupt priority that can be used by any interrupt service
104  * routine that makes calls to interrupt safe FreeRTOS API functions. DO NOT CALL
105  * INTERRUPT SAFE FREERTOS API FUNCTIONS FROM ANY INTERRUPT THAT HAS A HIGHER
106  * PRIORITY THAN THIS! (higher priorities are lower numeric values. */
107 #define configLIBRARY_MAX_SYSCALL_INTERRUPT_PRIORITY 5

109 /* Interrupt priorities used by the kernel port layer itself. These are generic
110 * to all Cortex-M ports, and do not rely on any particular library functions. */
111 #define configKERNEL_INTERRUPT_PRIORITY \
112  ( configLIBRARY_LOWEST_INTERRUPT_PRIORITY << ( 8 - configPRIO_BITS ) )
