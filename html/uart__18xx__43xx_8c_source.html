95 /* Initializes the pUART peripheral */
96 void Chip_UART_Init(LPC_USART_T *pUART)
97 {
98  volatile uint32_t tmp;
99 
100  /* Enable UART clocking. UART base clock(s) must already be enabled */
101  Chip_Clock_EnableOpts(UART_PClock[Chip_UART_GetIndex(pUART)], true, true, 1);
102 
103  /* Enable FIFOs by default, reset them */
104  Chip_UART_SetupFIFOS(pUART, (UART_FCR_FIFO_EN | UART_FCR_RX_RS | UART_FCR_TX_RS));
105 
106  /* Disable Tx */
107  Chip_UART_TXDisable(pUART);
108 
109  /* Disable interrupts */
110  pUART->IER = 0;
111  /* Set LCR to default state */
112  pUART->LCR = 0;
113  /* Set ACR to default state */
114  pUART->ACR = 0;
115  /* Set RS485 control to default state */
116  pUART->RS485CTRL = 0;
117  /* Set RS485 delay timer to default state */
118  pUART->RS485DLY = 0;
119  /* Set RS485 addr match to default state */
120  pUART->RS485ADRMATCH = 0;
121 
122  /* Clear MCR */
123  if (pUART == LPC_UART1) {
124  /* Set Modem Control to default state */
125  pUART->MCR = 0;
126  /*Dummy Reading to Clear Status */
127  tmp = pUART->MSR;
128  }
129 
130  /* Default 8N1, with DLAB disabled */
131  Chip_UART_ConfigData(pUART, (UART_LCR_WLEN8 | UART_LCR_SBS_1BIT | UART_LCR_PARITY_DIS));
132 
133  /* Disable fractional divider */
134  pUART->FDR = 0x10;
135 
136  (void) tmp;
137 }

151 {
152  int sent = 0;
153  uint8_t *p8 = (uint8_t *) data;
154 
155  /* Send until the transmit FIFO is full or out of bytes */
156  while ((sent < numBytes) &&
157  ((Chip_UART_ReadLineStatus(pUART) & UART_LSR_THRE) != 0)) {
158  Chip_UART_SendByte(pUART, *p8);
159  p8++;
160  sent++;
161  }

226 /* Determines and sets best dividers to get a target bit rate */
227 uint32_t Chip_UART_SetBaud(LPC_USART_T *pUART, uint32_t baudrate)
228 {
229  uint32_t div, divh, divl, clkin;
230 
231  /* Determine UART clock in rate without FDR */
232  clkin = Chip_Clock_GetRate(UART_BClock[Chip_UART_GetIndex(pUART)]);
233  div = clkin / (baudrate * 16);
234 
235  /* High and low halves of the divider */
236  divh = div / 256;
237  divl = div - (divh * 256);
238 
239  Chip_UART_EnableDivisorAccess(pUART);
240  Chip_UART_SetDivisorLatches(pUART, divl, divh);
241  Chip_UART_DisableDivisorAccess(pUART);
242 
243  /* Fractional FDR alreadt setup for 1 in UART init */
244 
245  return (clkin / div) >> 4;
246 }

259 void Chip_UART_TXIntHandlerRB(LPC_USART_T *pUART, RINGBUFF_T *pRB)

265  RingBuffer_Pop(pRB, &ch)) {
266  Chip_UART_SendByte(pUART, ch);
267  }
268

276 /* Populate a transmit ring buffer and start UART transmit */
277 uint32_t Chip_UART_SendRB(LPC_USART_T *pUART, RINGBUFF_T *pRB, const void *data, int bytes)
278 {
279  uint32_t ret;
280  uint8_t *p8 = (uint8_t *) data;
281 
282  /* Don't let UART transmit ring buffer change in the UART IRQ handler */
283  Chip_UART_IntDisable(pUART, UART_IER_THREINT);
284 
285  /* Move as much data as possible into transmit ring buffer */
286  ret = RingBuffer_InsertMult(pRB, p8, bytes);
287  Chip_UART_TXIntHandlerRB(pUART, pRB);
288 
289  /* Add additional data to transmit ring buffer if possible */
290  ret += RingBuffer_InsertMult(pRB, (p8 + ret), (bytes - ret));
291 
292  /* Enable UART transmit interrupt */
293  Chip_UART_IntEnable(pUART, UART_IER_THREINT);
294 
295  return ret;
296 }

326 /* Determines and sets best dividers to get a target baud rate */
327 uint32_t Chip_UART_SetBaudFDR(LPC_USART_T *pUART, uint32_t baud)
328 {
329  uint32_t sdiv = 0, sm = 1, sd = 0;
330  uint32_t pclk, m, d;
331  uint32_t odiff = -1UL; /* old best diff */
332 
333  /* Get base clock for the corresponding UART */
334  pclk = Chip_Clock_GetRate(UART_BClock[Chip_UART_GetIndex(pUART)]);
335 
336  /* Loop through all possible fractional divider values */
337  for (m = 1; odiff && m < 16; m++) {
338  for (d = 0; d < m; d++) {
339  uint32_t diff, div;
340  uint64_t dval = (((uint64_t) pclk << 28) * m) / (baud * (m + d));
341 
342  /* Lower 32-bit of dval has diff */
343  diff = (uint32_t) dval;
344  /* Upper 32-bit of dval has div */
345  div = (uint32_t) (dval >> 32);
346 
347  /* Closer to next div */
348  if ((int)diff < 0) {
349  diff = -diff;
350  div ++;
351  }
352 
353  /* Check if new value is worse than old or out of range */
354  if (odiff < diff || !div || (div >> 16) || (div < 3 && d)) {
355  continue;
356  }
357 
358  /* Store the new better values */
359  sdiv = div;
360  sd = d;
361  sm = m;
362  odiff = diff;
363 
364  /* On perfect match, break loop */
365  if(!diff) {
366  break;
367  }
368  }
369  }
370 
371  /* Return 0 if a vaild divisor is not possible */
372  if (!sdiv) {
373  return 0;
374  }
375 
376  /* Update UART registers */
377  Chip_UART_EnableDivisorAccess(pUART);
378  Chip_UART_SetDivisorLatches(pUART, UART_LOAD_DLL(sdiv), UART_LOAD_DLM(sdiv));
379  Chip_UART_DisableDivisorAccess(pUART);
380 
381  /* Set best fractional divider */
382  pUART->FDR = (UART_FDR_MULVAL(sm) | UART_FDR_DIVADDVAL(sd));
383 
384  /* Return actual baud rate */
385  return (pclk >> 4) * sm / (sdiv * (sm + sd));
386 }

395 /* Start/Stop Auto Baudrate activity */
396 void Chip_UART_ABCmd(LPC_USART_T *pUART, uint32_t mode, bool autorestart, FunctionalState NewState)
397 {
398  uint32_t tmp = 0;
399 
400  if (NewState == ENABLE) {
401  /* Clear DLL and DLM value */
402  pUART->LCR |= UART_LCR_DLAB_EN;
403  pUART->DLL = 0;
404  pUART->DLM = 0;
405  pUART->LCR &= ~UART_LCR_DLAB_EN;
406 
407  /* FDR value must be reset to default value */
408  pUART->FDR = 0x10;
409 
410  if (mode == UART_ACR_MODE1) {
411  tmp = UART_ACR_START | UART_ACR_MODE;
412  }
413  else {
414  tmp = UART_ACR_START;
415  }
416 
417  if (autorestart == true) {
418  tmp |= UART_ACR_AUTO_RESTART;
419  }
420  pUART->ACR = tmp;
421  }
422  else {
423  pUART->ACR = 0;
424  }
425 }
