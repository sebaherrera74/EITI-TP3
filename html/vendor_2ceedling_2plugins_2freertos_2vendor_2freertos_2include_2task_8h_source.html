1819 /*
1820  * THIS FUNCTION MUST NOT BE USED FROM APPLICATION CODE. IT IS ONLY
1821  * INTENDED FOR USE WHEN IMPLEMENTING A PORT OF THE SCHEDULER AND IS
1822  * AN INTERFACE WHICH IS FOR THE EXCLUSIVE USE OF THE SCHEDULER.
1823  *
1824  * Called from the real time kernel tick (either preemptive or cooperative),
1825  * this increments the tick count and checks if any tasks that are blocked
1826  * for a finite period required removing from a blocked list and placing on
1827  * a ready list. If a non-zero value is returned then a context switch is
1828  * required because either:
1829  * + A task was removed from a blocked list because its timeout had expired,
1830  * or
1831  * + Time slicing is in use and there is a task of equal priority to the
1832  * currently running task.
1833  */
1834 BaseType_t xTaskIncrementTick( void ) PRIVILEGED_FUNCTION;

1836 /*
1837  * THIS FUNCTION MUST NOT BE USED FROM APPLICATION CODE. IT IS AN
1838  * INTERFACE WHICH IS FOR THE EXCLUSIVE USE OF THE SCHEDULER.
1839  *
1840  * THIS FUNCTION MUST BE CALLED WITH INTERRUPTS DISABLED.
1841  *
1842  * Removes the calling task from the ready list and places it both
1843  * on the list of tasks waiting for a particular event, and the
1844  * list of delayed tasks. The task will be removed from both lists
1845  * and replaced on the ready list should either the event occur (and
1846  * there be no higher priority tasks waiting on the same event) or
1847  * the delay period expires.
1848  *
1849  * The 'unordered' version replaces the event list item value with the
1850  * xItemValue value, and inserts the list item at the end of the list.
1851  *
1852  * The 'ordered' version uses the existing event list item value (which is the
1853  * owning tasks priority) to insert the list item into the event list is task
1854  * priority order.
1855  *
1856  * @param pxEventList The list containing tasks that are blocked waiting
1857  * for the event to occur.
1858  *
1859  * @param xItemValue The item value to use for the event list item when the
1860  * event list is not ordered by task priority.
1861  *
1862  * @param xTicksToWait The maximum amount of time that the task should wait
1863  * for the event to occur. This is specified in kernel ticks,the constant
1864  * portTICK_PERIOD_MS can be used to convert kernel ticks into a real time
1865  * period.
1866  */
1867 void vTaskPlaceOnEventList( List_t * const pxEventList, const TickType_t xTicksToWait ) PRIVILEGED_FUNCTION;
1868 void vTaskPlaceOnUnorderedEventList( List_t * pxEventList, const TickType_t xItemValue, const TickType_t xTicksToWait ) PRIVILEGED_FUNCTION;
