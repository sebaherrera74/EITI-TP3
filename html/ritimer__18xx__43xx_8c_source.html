50 /* Initialize the RIT */
51 void Chip_RIT_Init(LPC_RITIMER_T *pRITimer)
52 {
53  Chip_Clock_EnableOpts(CLK_MX_RITIMER, true, true, 1);
54  pRITimer->COMPVAL = 0xFFFFFFFF;
55  pRITimer->MASK = 0x00000000;

67 /* Set timer interval value */
68 void Chip_RIT_SetTimerInterval(LPC_RITIMER_T *pRITimer, uint32_t time_interval)
69 {
70  uint32_t cmp_value;
71 
72  /* Determine aapproximate compare value based on clock rate and passed interval */
73  cmp_value = (Chip_Clock_GetRate(CLK_MX_RITIMER) / 1000) * time_interval;
74 
75  /* Set timer compare value */
76  Chip_RIT_SetCOMPVAL(pRITimer, cmp_value);
77 
78  /* Set timer enable clear bit to clear timer to 0 whenever
79  counter value equals the contents of RICOMPVAL */
80  Chip_RIT_EnableCTRL(pRITimer, RIT_CTRL_ENCLR);
81 }

83 /* Check whether interrupt is pending */
84 IntStatus Chip_RIT_GetIntStatus(LPC_RITIMER_T *pRITimer)
85 {
86  uint8_t result;
87 
88  if ((pRITimer->CTRL & RIT_CTRL_INT) == 1) {
89  result = SET;

91  else {
92  return RESET;
93  }
94 
95  return (IntStatus) result;
96 }
