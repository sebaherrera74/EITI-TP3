124 void Board_SetupClocking(void)
125 {
126  Chip_CREG_SetFlashAcceleration(MAX_CLOCK_FREQ);
127  Chip_SetupCoreClock(CLKIN_CRYSTAL, MAX_CLOCK_FREQ, true);
128 
129  /* Setup system base clocks and initial states. This won't enable and
130  disable individual clocks, but sets up the base clock sources for
131  each individual peripheral clock. */
132  for (uint32_t i = 0; i < (sizeof(InitClkStates) / sizeof(InitClkStates[0])); ++i)
133  {
134  const struct CLK_BASE_STATES *c = &InitClkStates[i];
135  Chip_Clock_SetBaseClock(c->clk, c->clkin, c->autoblock_enab,
136  c->powerdn);
137  }
138 
139  /* Reset and enable 32Khz oscillator */
140  LPC_CREG->CREG0 &= ~((1 << 3) | (1 << 2));
141  LPC_CREG->CREG0 |= (1 << 1) | (1 << 0);
142 }

144 /* Set up and initialize hardware prior to call to main */
145 void Board_SystemInit(void)
146 {
147  /* Setup system clocking and memory. This is done early to allow the
148  application and tools to clear memory and use scatter loading to
149  external memory. */
150  Board_SetupMuxing();
151  Board_SetupClocking();
152  SystemCoreClockUpdate();
153 }
