191 StackType_t * pxPortInitialiseStack( StackType_t * pxTopOfStack,
192  TaskFunction_t pxCode,
193  void * pvParameters,
194  BaseType_t xRunPrivileged )
195 {
196  /* Simulate the stack frame as it would be created by a context switch
197  * interrupt. */
198  pxTopOfStack--; /* Offset added to account for the way the MCU uses the stack on entry/exit of interrupts. */
199  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
200  pxTopOfStack--;
201  *pxTopOfStack = ( ( StackType_t ) pxCode ) & portSTART_ADDRESS_MASK; /* PC */
202  pxTopOfStack--;
203  *pxTopOfStack = 0; /* LR */
204  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
205  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
206  pxTopOfStack -= 9; /* R11, R10, R9, R8, R7, R6, R5 and R4. */
207 
208  if( xRunPrivileged == pdTRUE )
209  {
210  *pxTopOfStack = portINITIAL_CONTROL_IF_PRIVILEGED;
211  }
212  else
213  {
214  *pxTopOfStack = portINITIAL_CONTROL_IF_UNPRIVILEGED;
215  }
216 
217  return pxTopOfStack;
218 }

240 static void prvSVCHandler( uint32_t * pulParam )
241 {
242  uint8_t ucSVCNumber;
243  uint32_t ulPC;
244 
245  #if ( configENFORCE_SYSTEM_CALLS_FROM_KERNEL_ONLY == 1 )
246  #if defined( __ARMCC_VERSION )
247 
248  /* Declaration when these variable are defined in code instead of being
249  * exported from linker scripts. */
250  extern uint32_t * __syscalls_flash_start__;
251  extern uint32_t * __syscalls_flash_end__;
252  #else
253  /* Declaration when these variable are exported from linker scripts. */
254  extern uint32_t __syscalls_flash_start__[];
255  extern uint32_t __syscalls_flash_end__[];
256  #endif /* #if defined( __ARMCC_VERSION ) */
257  #endif /* #if( configENFORCE_SYSTEM_CALLS_FROM_KERNEL_ONLY == 1 ) */
258 
259  /* The stack contains: r0, r1, r2, r3, r12, LR, PC and xPSR. The first
260  * argument (r0) is pulParam[ 0 ]. */
261  ulPC = pulParam[ portOFFSET_TO_PC ];
262  ucSVCNumber = ( ( uint8_t * ) ulPC )[ -2 ];
263 
264  switch( ucSVCNumber )
265  {
266  case portSVC_START_SCHEDULER:
267  portNVIC_SHPR2_REG |= portNVIC_SVC_PRI;
268  prvRestoreContextOfFirstTask();
269  break;
270 
271  case portSVC_YIELD:
272  portNVIC_INT_CTRL_REG = portNVIC_PENDSVSET_BIT;
273 
274  /* Barriers are normally not required
275  * but do ensure the code is completely
276  * within the specified behaviour for the
277  * architecture. */
278  __asm volatile ( "dsb" ::: "memory" );
279  __asm volatile ( "isb" );
280 
281  break;
282 
283 
284  #if ( configENFORCE_SYSTEM_CALLS_FROM_KERNEL_ONLY == 1 )
285  case portSVC_RAISE_PRIVILEGE: /* Only raise the privilege, if the
286  * svc was raised from any of the
287  * system calls. */
288 
289  if( ( ulPC >= ( uint32_t ) __syscalls_flash_start__ ) &&
290  ( ulPC <= ( uint32_t ) __syscalls_flash_end__ ) )
291  {
292  __asm volatile
293  (
294  " mrs r1, control \n"/* Obtain current control value. */
295  " bic r1, #1 \n"/* Set privilege bit. */
296  " msr control, r1 \n"/* Write back new control value. */
297  ::: "r1", "memory"
298  );
299  }
300 
301  break;
302  #else /* if ( configENFORCE_SYSTEM_CALLS_FROM_KERNEL_ONLY == 1 ) */
303  case portSVC_RAISE_PRIVILEGE:
304  __asm volatile
305  (
306  " mrs r1, control \n"/* Obtain current control value. */
307  " bic r1, #1 \n"/* Set privilege bit. */
308  " msr control, r1 \n"/* Write back new control value. */
309  ::: "r1", "memory"
310  );
311  break;
312  #endif /* #if( configENFORCE_SYSTEM_CALLS_FROM_KERNEL_ONLY == 1 ) */
313 
314  default: /* Unknown SVC call. */
315  break;
316  }
317 }

320 static void prvRestoreContextOfFirstTask( void )
321 {
322  __asm volatile
323  (
324  " ldr r0, =0xE000ED08 \n"/* Use the NVIC offset register to locate the stack. */
325  " ldr r0, [r0] \n"
326  " ldr r0, [r0] \n"
327  " msr msp, r0 \n"/* Set the msp back to the start of the stack. */
328  " ldr r3, pxCurrentTCBConst2 \n"/* Restore the context. */
329  " ldr r1, [r3] \n"
330  " ldr r0, [r1] \n"/* The first item in the TCB is the task top of stack. */
331  " add r1, r1, #4 \n"/* Move onto the second item in the TCB... */
332  " \n"
333  " dmb \n"/* Complete outstanding transfers before disabling MPU. */
334  " ldr r2, =0xe000ed94 \n"/* MPU_CTRL register. */
335  " ldr r3, [r2] \n"/* Read the value of MPU_CTRL. */
336  " bic r3, #1 \n"/* r3 = r3 & ~1 i.e. Clear the bit 0 in r3. */
337  " str r3, [r2] \n"/* Disable MPU. */
338  " \n"
339  " ldr r2, =0xe000ed9c \n"/* Region Base Address register. */
340  " ldmia r1!, {r4-r11} \n"/* Read 4 sets of MPU registers. */
341  " stmia r2!, {r4-r11} \n"/* Write 4 sets of MPU registers. */
342  " \n"
343  " ldr r2, =0xe000ed94 \n"/* MPU_CTRL register. */
344  " ldr r3, [r2] \n"/* Read the value of MPU_CTRL. */
345  " orr r3, #1 \n"/* r3 = r3 | 1 i.e. Set the bit 0 in r3. */
346  " str r3, [r2] \n"/* Enable MPU. */
347  " dsb \n"/* Force memory writes before continuing. */
348  " \n"
349  " ldmia r0!, {r3, r4-r11} \n"/* Pop the registers that are not automatically saved on exception entry. */
350  " msr control, r3 \n"
351  " msr psp, r0 \n"/* Restore the task stack pointer. */
352  " mov r0, #0 \n"
353  " msr basepri, r0 \n"
354  " ldr r14, =0xfffffffd \n"/* Load exec return code. */
355  " bx r14 \n"
356  " \n"
357  " .align 4 \n"
358  "pxCurrentTCBConst2: .word pxCurrentTCB \n"
359  );
360 }

366 BaseType_t xPortStartScheduler( void )
367 {
368  /* configMAX_SYSCALL_INTERRUPT_PRIORITY must not be set to 0. See
369  * https://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html */
370  configASSERT( ( configMAX_SYSCALL_INTERRUPT_PRIORITY ) );
371 
372  #if ( configASSERT_DEFINED == 1 )
373  {
374  volatile uint32_t ulOriginalPriority;
375  volatile uint8_t * const pucFirstUserPriorityRegister = ( volatile uint8_t * const ) ( portNVIC_IP_REGISTERS_OFFSET_16 + portFIRST_USER_INTERRUPT_NUMBER );
376  volatile uint8_t ucMaxPriorityValue;
377 
378  /* Determine the maximum priority from which ISR safe FreeRTOS API
379  * functions can be called. ISR safe functions are those that end in
380  * "FromISR". FreeRTOS maintains separate thread and ISR API functions
381  * to ensure interrupt entry is as fast and simple as possible.
382  *
383  * Save the interrupt priority value that is about to be clobbered. */
384  ulOriginalPriority = *pucFirstUserPriorityRegister;
385 
386  /* Determine the number of priority bits available. First write to all
387  * possible bits. */
388  *pucFirstUserPriorityRegister = portMAX_8_BIT_VALUE;
389 
390  /* Read the value back to see how many bits stuck. */
391  ucMaxPriorityValue = *pucFirstUserPriorityRegister;
392 
393  /* Use the same mask on the maximum system call priority. */
394  ucMaxSysCallPriority = configMAX_SYSCALL_INTERRUPT_PRIORITY & ucMaxPriorityValue;
395 
396  /* Calculate the maximum acceptable priority group value for the number
397  * of bits read back. */
398  ulMaxPRIGROUPValue = portMAX_PRIGROUP_BITS;
399 
400  while( ( ucMaxPriorityValue & portTOP_BIT_OF_BYTE ) == portTOP_BIT_OF_BYTE )
401  {
402  ulMaxPRIGROUPValue--;
403  ucMaxPriorityValue <<= ( uint8_t ) 0x01;
404  }
405 
406  #ifdef __NVIC_PRIO_BITS
407  {
408  /* Check the CMSIS configuration that defines the number of
409  * priority bits matches the number of priority bits actually queried
410  * from the hardware. */
411  configASSERT( ( portMAX_PRIGROUP_BITS - ulMaxPRIGROUPValue ) == __NVIC_PRIO_BITS );
412  }
413  #endif
414 
415  #ifdef configPRIO_BITS
416  {
417  /* Check the FreeRTOS configuration that defines the number of
418  * priority bits matches the number of priority bits actually queried
419  * from the hardware. */
420  configASSERT( ( portMAX_PRIGROUP_BITS - ulMaxPRIGROUPValue ) == configPRIO_BITS );
421  }
422  #endif
423 
424  /* Shift the priority group value back to its position within the AIRCR
425  * register. */
426  ulMaxPRIGROUPValue <<= portPRIGROUP_SHIFT;
427  ulMaxPRIGROUPValue &= portPRIORITY_GROUP_MASK;
428 
429  /* Restore the clobbered interrupt priority register to its original
430  * value. */
431  *pucFirstUserPriorityRegister = ulOriginalPriority;
432  }
433  #endif /* conifgASSERT_DEFINED */
434 
435  /* Make PendSV and SysTick the same priority as the kernel, and the SVC
436  * handler higher priority so it can be used to exit a critical section (where
437  * lower priorities are masked). */
438  portNVIC_SHPR3_REG |= portNVIC_PENDSV_PRI;
439  portNVIC_SHPR3_REG |= portNVIC_SYSTICK_PRI;
440 
441  /* Configure the regions in the MPU that are common to all tasks. */
442  prvSetupMPU();
443 
444  /* Start the timer that generates the tick ISR. Interrupts are disabled
445  * here already. */
446  vPortSetupTimerInterrupt();
447 
448  /* Initialise the critical nesting count ready for the first task. */
449  uxCriticalNesting = 0;
450 
451  /* Start the first task. */
452  __asm volatile (
453  " ldr r0, =0xE000ED08 \n"/* Use the NVIC offset register to locate the stack. */
454  " ldr r0, [r0] \n"
455  " ldr r0, [r0] \n"
456  " msr msp, r0 \n"/* Set the msp back to the start of the stack. */
457  " cpsie i \n"/* Globally enable interrupts. */
458  " cpsie f \n"
459  " dsb \n"
460  " isb \n"
461  " svc %0 \n"/* System call to start first task. */
462  " nop \n"
463  " .ltorg \n"
464  ::"i" ( portSVC_START_SCHEDULER ) : "memory" );
465 
466  /* Should not get here! */
467  return 0;
468 }

662 static uint32_t prvGetMPURegionSizeSetting( uint32_t ulActualSizeInBytes )
663 {
664  uint32_t ulRegionSize, ulReturnValue = 4;
665 
666  /* 32 is the smallest region size, 31 is the largest valid value for
667  * ulReturnValue. */
668  for( ulRegionSize = 32UL; ulReturnValue < 31UL; ( ulRegionSize <<= 1UL ) )
669  {
670  if( ulActualSizeInBytes <= ulRegionSize )
671  {
672  break;
673  }
674  else
675  {
676  ulReturnValue++;
677  }
678  }
679 
680  /* Shift the code by one before returning so it can be written directly
681  * into the the correct bit position of the attribute register. */
682  return( ulReturnValue << 1UL );
683 }

686 BaseType_t xIsPrivileged( void ) /* __attribute__ (( naked )) */
687 {
688  __asm volatile
689  (
690  " mrs r0, control \n"/* r0 = CONTROL. */
691  " tst r0, #1 \n"/* Perform r0 & 1 (bitwise AND) and update the conditions flag. */
692  " ite ne \n"
693  " movne r0, #0 \n"/* CONTROL[0]!=0. Return false to indicate that the processor is not privileged. */

696  " \n"
697  " .align 4 \n"
698  ::: "r0", "memory"
699  );

716 void vPortStoreTaskMPUSettings( xMPU_SETTINGS * xMPUSettings,
717  const struct xMEMORY_REGION * const xRegions,
718  StackType_t * pxBottomOfStack,
719  uint32_t ulStackDepth )
720 {

722  extern uint32_t __SRAM_segment_end__[];
723  extern uint32_t __privileged_data_start__[];
724  extern uint32_t __privileged_data_end__[];

726  uint32_t ul;
727 
728  if( xRegions == NULL )
729  {
730  /* No MPU regions are specified so allow access to all RAM. */
731  xMPUSettings->xRegion[ 0 ].ulRegionBaseAddress =
732  ( ( uint32_t ) __SRAM_segment_start__ ) | /* Base address. */
733  ( portMPU_REGION_VALID ) |

737  ( portMPU_REGION_READ_WRITE ) |
738  ( portMPU_REGION_CACHEABLE_BUFFERABLE ) |

740  ( portMPU_REGION_ENABLE );
741 
742  /* Re-instate the privileged only RAM region as xRegion[ 0 ] will have
743  * just removed the privileged only parameters. */
744  xMPUSettings->xRegion[ 1 ].ulRegionBaseAddress =
745  ( ( uint32_t ) __privileged_data_start__ ) | /* Base address. */
746  ( portMPU_REGION_VALID ) |
747  ( portSTACK_REGION + 1 );
748 
749  xMPUSettings->xRegion[ 1 ].ulRegionAttribute =
750  ( portMPU_REGION_PRIVILEGED_READ_WRITE ) |
751  ( portMPU_REGION_CACHEABLE_BUFFERABLE ) |
752  prvGetMPURegionSizeSetting( ( uint32_t ) __privileged_data_end__ - ( uint32_t ) __privileged_data_start__ ) |
753  ( portMPU_REGION_ENABLE );
754 
755  /* Invalidate all other regions. */
756  for( ul = 2; ul <= portNUM_CONFIGURABLE_REGIONS; ul++ )
757  {
758  xMPUSettings->xRegion[ ul ].ulRegionBaseAddress = ( portSTACK_REGION + ul ) | portMPU_REGION_VALID;
759  xMPUSettings->xRegion[ ul ].ulRegionAttribute = 0UL;
760  }
761  }
762  else
763  {
764  /* This function is called automatically when the task is created - in
765  * which case the stack region parameters will be valid. At all other
766  * times the stack parameters will not be valid and it is assumed that the
767  * stack region has already been configured. */
768  if( ulStackDepth > 0 )
769  {
770  /* Define the region that allows access to the stack. */
771  xMPUSettings->xRegion[ 0 ].ulRegionBaseAddress =
772  ( ( uint32_t ) pxBottomOfStack ) |
773  ( portMPU_REGION_VALID ) |
774  ( portSTACK_REGION ); /* Region number. */
775 
776  xMPUSettings->xRegion[ 0 ].ulRegionAttribute =
777  ( portMPU_REGION_READ_WRITE ) | /* Read and write. */
778  ( prvGetMPURegionSizeSetting( ulStackDepth * ( uint32_t ) sizeof( StackType_t ) ) ) |
779  ( portMPU_REGION_CACHEABLE_BUFFERABLE ) |
780  ( portMPU_REGION_ENABLE );
781  }
782 
783  lIndex = 0;
784 
785  for( ul = 1; ul <= portNUM_CONFIGURABLE_REGIONS; ul++ )
786  {
787  if( ( xRegions[ lIndex ] ).ulLengthInBytes > 0UL )
788  {
789  /* Translate the generic region definition contained in
790  * xRegions into the CM3 specific MPU settings that are then
791  * stored in xMPUSettings. */
792  xMPUSettings->xRegion[ ul ].ulRegionBaseAddress =
793  ( ( uint32_t ) xRegions[ lIndex ].pvBaseAddress ) |
794  ( portMPU_REGION_VALID ) |
795  ( portSTACK_REGION + ul ); /* Region number. */
796 
797  xMPUSettings->xRegion[ ul ].ulRegionAttribute =
798  ( prvGetMPURegionSizeSetting( xRegions[ lIndex ].ulLengthInBytes ) ) |
799  ( xRegions[ lIndex ].ulParameters ) |
800  ( portMPU_REGION_ENABLE );
801  }
802  else
803  {
804  /* Invalidate the region. */
805  xMPUSettings->xRegion[ ul ].ulRegionBaseAddress = ( portSTACK_REGION + ul ) | portMPU_REGION_VALID;
806  xMPUSettings->xRegion[ ul ].ulRegionAttribute = 0UL;
807  }
808 
809  lIndex++;
810  }
811  }

817  void vPortValidateInterruptPriority( void )
818  {
819  uint32_t ulCurrentInterrupt;
820  uint8_t ucCurrentPriority;
821 
822  /* Obtain the number of the currently executing interrupt. */
823  __asm volatile ( "mrs %0, ipsr" : "=r" ( ulCurrentInterrupt )::"memory" );
824 
825  /* Is the interrupt number a user defined interrupt? */
826  if( ulCurrentInterrupt >= portFIRST_USER_INTERRUPT_NUMBER )
827  {
828  /* Look up the interrupt's priority. */
829  ucCurrentPriority = pcInterruptPriorityRegisters[ ulCurrentInterrupt ];
830 
831  /* The following assertion will fail if a service routine (ISR) for
832  * an interrupt that has been assigned a priority above
833  * configMAX_SYSCALL_INTERRUPT_PRIORITY calls an ISR safe FreeRTOS API
834  * function. ISR safe FreeRTOS API functions must *only* be called
835  * from interrupts that have been assigned a priority at or below
836  * configMAX_SYSCALL_INTERRUPT_PRIORITY.
837  *
838  * Numerically low interrupt priority numbers represent logically high
839  * interrupt priorities, therefore the priority of the interrupt must
840  * be set to a value equal to or numerically *higher* than
841  * configMAX_SYSCALL_INTERRUPT_PRIORITY.
842  *
843  * Interrupts that use the FreeRTOS API must not be left at their
844  * default priority of zero as that is the highest possible priority,
845  * which is guaranteed to be above configMAX_SYSCALL_INTERRUPT_PRIORITY,
846  * and therefore also guaranteed to be invalid.
847  *
848  * FreeRTOS maintains separate thread and ISR API functions to ensure
849  * interrupt entry is as fast and simple as possible.
850  *
851  * The following links provide detailed information:
852  * https://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html
853  * https://www.FreeRTOS.org/FAQHelp.html */
854  configASSERT( ucCurrentPriority >= ucMaxSysCallPriority );
855  }
856 
857  /* Priority grouping: The interrupt controller (NVIC) allows the bits
858  * that define each interrupt's priority to be split between bits that
859  * define the interrupt's pre-emption priority bits and bits that define
860  * the interrupt's sub-priority. For simplicity all bits must be defined
861  * to be pre-emption priority bits. The following assertion will fail if
862  * this is not the case (if some bits represent a sub-priority).
863  *
864  * If the application only uses CMSIS libraries for interrupt
865  * configuration then the correct setting can be achieved on all Cortex-M
866  * devices by calling NVIC_SetPriorityGrouping( 0 ); before starting the
867  * scheduler. Note however that some vendor specific peripheral libraries
868  * assume a non-zero priority group setting, in which cases using a value
869  * of zero will result in unpredicable behaviour. */
870  configASSERT( ( portAIRCR_REG & portPRIORITY_GROUP_MASK ) <= ulMaxPRIGROUPValue );
871  }
