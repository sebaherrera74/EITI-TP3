31  ----------------------------------------------------------------------------
32 
33  This source is a new LPCOpen board port for Edu-CIAA.
34 
35  It exposes the same API found in board_api.h, along with new functions for
36  input handling (TEC_1 to TEC_4 buttons), getting ADC samples and such.
37 
38  Most functionality exposed on P1/P2 headers are initialized and configured
39  based on original intend/label: ETHERNET, 232_RX/TX, SPI, GPIO, I2C & ADC.
40 
41  GPIO pins are configured as INPUT with pullups as if they were buttons.
42 
43  ETHERNET port requires a specific PONCHO with an ethernet Physical layer
44  (PHY). As a result, there is no code ported to initialize that IC as it
45  could be any. All there is here is MUX configuration to enable RMII and
46  start the peripheral clock.
47 
48  External hardware is also required for CAN, 4 bit Character display (LCD
49  port) and keyboard (port T_*).
50 
51  DEBUG UART (wired to FT2232 IC and exposed on the development PC as a
52  USB serial port) is configured with this default settings: 115200, 8N1.
53 
54  But -as it happens with I2C, SPI, ADC, ect- its defaults can be configured
55  at will before including board.h (please look at that file to know the
56  corresponding #defines)
57 
58  Pin MUXING is configured on board_sysinit.c. While it is known the intend of
59  that file is to configure a SDRAM interface before main(), I prefer to have
60  the amount of mux configurations enclosed on that file alone, letting this
61  one cleaner.

250 void Board_ADC_ReadBegin(ADC_CHANNEL_T channel)
251 {
252  if (channel < ADC_CH0 || channel > ADC_CH3)
253  {
254  return;
255  }
256 
257  if (curADCChannel >= ADC_CH0 || curADCChannel <= ADC_CH3)
258  {
259  Chip_ADC_EnableChannel(LPC_ADC0, curADCChannel, DISABLE);
260  }
261 
262  curADCChannel = channel;
263 
264  Chip_SCU_ADC_Channel_Config(0, channel);
265  Chip_ADC_EnableChannel(LPC_ADC0, channel, ENABLE);
266  Chip_ADC_SetBurstCmd(LPC_ADC0, DISABLE);
267  Chip_ADC_SetStartMode(LPC_ADC0, ADC_START_NOW, ADC_TRIGGERMODE_RISING);
268 }
269 
270 bool Board_ADC_ReadWait()
271 {
272  return (Chip_ADC_ReadStatus(LPC_ADC0, curADCChannel, ADC_DR_DONE_STAT) == RESET);
273 }
274 
275 uint16_t Board_ADC_ReadEnd()
276 {
277  uint16_t data;
278 
279  if (Chip_ADC_ReadValue(LPC_ADC0, curADCChannel, &data) != SUCCESS)
280  {
281  data = 0xFFFF;
282  }
283 
284  Chip_ADC_EnableChannel(LPC_ADC0, curADCChannel, DISABLE);
285  curADCChannel = 0xFF;
286  return data;
287 }
