191 void Board_SetupMuxing(void) {
192  for (int index = 0; index < sizeof(INIT_PIN_MUXING) / sizeof(gpio_pins_config_t); index++ ) {
193  const gpio_pins_config_t * config = &INIT_PIN_MUXING[index];
194  Chip_SCU_PinMuxSet(config->port, config->pin, config->mode);
195  Chip_GPIO_SetPinState(LPC_GPIO_PORT, config->gpio, config->bit, false);
196  Chip_GPIO_SetPinDIR(LPC_GPIO_PORT, config->gpio, config->bit, config->output);
197  }
198 }
199 
200 void Board_SetupClocking(void) {
201  Chip_CREG_SetFlashAcceleration(MAX_CLOCK_FREQ);
202  Chip_SetupCoreClock(CLKIN_CRYSTAL, MAX_CLOCK_FREQ, true);
203 
204  /* Setup system base clocks and initial states. This won't enable and
205  disable individual clocks, but sets up the base clock sources for
206  each individual peripheral clock. */
207  for (uint32_t i = 0; i < (sizeof(INIT_CLK_STATES) / sizeof(clock_states_t)); ++i) {
208  const clock_states_t *c = &INIT_CLK_STATES[i];
209  Chip_Clock_SetBaseClock(c->clk, c->clkin, c->autoblock_enab,
210  c->powerdn);
211  }
212 
213  /* Reset and enable 32Khz oscillator */
214  LPC_CREG->CREG0 &= ~((1 << 3) | (1 << 2));
215  LPC_CREG->CREG0 |= (1 << 1) | (1 << 0);
216 }
