92 /*
93  * Insert a block into the list of free blocks - which is ordered by size of
94  * the block. Small blocks at the start of the list and large blocks at the end
95  * of the list.
96  */
97 #define prvInsertBlockIntoFreeList( pxBlockToInsert ) \
98  { \
99  BlockLink_t * pxIterator; \
100  size_t xBlockSize; \
101  \
102  xBlockSize = pxBlockToInsert->xBlockSize; \
103  \
104  /* Iterate through the list until a block is found that has a larger size */ \
105  /* than the block we are inserting. */ \
106  for( pxIterator = &xStart; pxIterator->pxNextFreeBlock->xBlockSize < xBlockSize; pxIterator = pxIterator->pxNextFreeBlock ) \
107  { \
108  /* There is nothing to do here - just iterate to the correct position. */ \
109  } \
110  \
111  /* Update the list to include the block being inserted in the correct */ \
112  /* position. */ \
113  pxBlockToInsert->pxNextFreeBlock = pxIterator->pxNextFreeBlock; \
114  pxIterator->pxNextFreeBlock = pxBlockToInsert; \
115  }

118 void * pvPortMalloc( size_t xWantedSize )
119 {
120  BlockLink_t * pxBlock, * pxPreviousBlock, * pxNewBlockLink;
121  static BaseType_t xHeapHasBeenInitialised = pdFALSE;
122  void * pvReturn = NULL;
123 
124  vTaskSuspendAll();
125  {
126  /* If this is the first call to malloc then the heap will require
127  * initialisation to setup the list of free blocks. */
128  if( xHeapHasBeenInitialised == pdFALSE )
129  {
130  prvHeapInit();
131  xHeapHasBeenInitialised = pdTRUE;
132  }
133 
134  /* The wanted size must be increased so it can contain a BlockLink_t
135  * structure in addition to the requested amount of bytes. */
136  if( ( xWantedSize > 0 ) &&
137  ( ( xWantedSize + heapSTRUCT_SIZE ) > xWantedSize ) ) /* Overflow check */
138  {
139  xWantedSize += heapSTRUCT_SIZE;
140 
141  /* Byte alignment required. Check for overflow. */
142  if( ( xWantedSize + ( portBYTE_ALIGNMENT - ( xWantedSize & portBYTE_ALIGNMENT_MASK ) ) )
143  > xWantedSize )
144  {
145  xWantedSize += ( portBYTE_ALIGNMENT - ( xWantedSize & portBYTE_ALIGNMENT_MASK ) );
146  configASSERT( ( xWantedSize & portBYTE_ALIGNMENT_MASK ) == 0 );
147  }
148  else
149  {
150  xWantedSize = 0;
151  }
152  }
153  else
154  {
155  xWantedSize = 0;
156  }
157 
158 
159  if( ( xWantedSize > 0 ) && ( xWantedSize <= xFreeBytesRemaining ) )
160  {
161  /* Blocks are stored in byte order - traverse the list from the start
162  * (smallest) block until one of adequate size is found. */
163  pxPreviousBlock = &xStart;
164  pxBlock = xStart.pxNextFreeBlock;
165 
166  while( ( pxBlock->xBlockSize < xWantedSize ) && ( pxBlock->pxNextFreeBlock != NULL ) )
167  {
168  pxPreviousBlock = pxBlock;
169  pxBlock = pxBlock->pxNextFreeBlock;
170  }
171 
172  /* If we found the end marker then a block of adequate size was not found. */
173  if( pxBlock != &xEnd )
174  {
175  /* Return the memory space - jumping over the BlockLink_t structure
176  * at its start. */
177  pvReturn = ( void * ) ( ( ( uint8_t * ) pxPreviousBlock->pxNextFreeBlock ) + heapSTRUCT_SIZE );
178 
179  /* This block is being returned for use so must be taken out of the
180  * list of free blocks. */
181  pxPreviousBlock->pxNextFreeBlock = pxBlock->pxNextFreeBlock;
182 
183  /* If the block is larger than required it can be split into two. */
184  if( ( pxBlock->xBlockSize - xWantedSize ) > heapMINIMUM_BLOCK_SIZE )
185  {
186  /* This block is to be split into two. Create a new block
187  * following the number of bytes requested. The void cast is
188  * used to prevent byte alignment warnings from the compiler. */
189  pxNewBlockLink = ( void * ) ( ( ( uint8_t * ) pxBlock ) + xWantedSize );
190 
191  /* Calculate the sizes of two blocks split from the single
192  * block. */
193  pxNewBlockLink->xBlockSize = pxBlock->xBlockSize - xWantedSize;
194  pxBlock->xBlockSize = xWantedSize;
195 
196  /* Insert the new block into the list of free blocks. */
197  prvInsertBlockIntoFreeList( ( pxNewBlockLink ) );
198  }
199 
200  xFreeBytesRemaining -= pxBlock->xBlockSize;
201  }
202  }
203 
204  traceMALLOC( pvReturn, xWantedSize );
205  }
206  ( void ) xTaskResumeAll();
207 
208  #if ( configUSE_MALLOC_FAILED_HOOK == 1 )
209  {
210  if( pvReturn == NULL )
211  {
212  extern void vApplicationMallocFailedHook( void );
213  vApplicationMallocFailedHook();
214  }
215  }
216  #endif
217 
218  return pvReturn;
219 }

222 void vPortFree( void * pv )
223 {
224  uint8_t * puc = ( uint8_t * ) pv;
225  BlockLink_t * pxLink;
226 
227  if( pv != NULL )
228  {
229  /* The memory being freed will have an BlockLink_t structure immediately
230  * before it. */
231  puc -= heapSTRUCT_SIZE;
232 
233  /* This unexpected casting is to keep some compilers from issuing
234  * byte alignment warnings. */
235  pxLink = ( void * ) puc;
236 
237  vTaskSuspendAll();
238  {
239  /* Add this block to the list of free blocks. */
240  prvInsertBlockIntoFreeList( ( ( BlockLink_t * ) pxLink ) );
241  xFreeBytesRemaining += pxLink->xBlockSize;
242  traceFREE( pv, pxLink->xBlockSize );
243  }
244  ( void ) xTaskResumeAll();
245  }
246 }

261 static void prvHeapInit( void )
262 {
263  BlockLink_t * pxFirstFreeBlock;
264  uint8_t * pucAlignedHeap;
265 
266  /* Ensure the heap starts on a correctly aligned boundary. */
267  pucAlignedHeap = ( uint8_t * ) ( ( ( portPOINTER_SIZE_TYPE ) & ucHeap[ portBYTE_ALIGNMENT ] ) & ( ~( ( portPOINTER_SIZE_TYPE ) portBYTE_ALIGNMENT_MASK ) ) );
268 
269  /* xStart is used to hold a pointer to the first item in the list of free
270  * blocks. The void cast is used to prevent compiler warnings. */
271  xStart.pxNextFreeBlock = ( void * ) pucAlignedHeap;
272  xStart.xBlockSize = ( size_t ) 0;
273 
274  /* xEnd is used to mark the end of the list of free blocks. */
275  xEnd.xBlockSize = configADJUSTED_HEAP_SIZE;
276  xEnd.pxNextFreeBlock = NULL;
277 
278  /* To start with there is a single free block that is sized to take up the
279  * entire heap space. */
280  pxFirstFreeBlock = ( void * ) pucAlignedHeap;
281  pxFirstFreeBlock->xBlockSize = configADJUSTED_HEAP_SIZE;
282  pxFirstFreeBlock->pxNextFreeBlock = &xEnd;
283 }
