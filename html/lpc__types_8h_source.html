111 /* _SBF(f,v) sets the bit field starting at position "f" to value "v".
112  * _SBF(f,v) is intended to be used in "OR" and "AND" expressions:
113  * e.g., "((_SBF(5,7) | _SBF(12,0xF)) & 0xFFFF)"
114  */

119 /* _BITMASK constructs a symbol with 'field_width' least significant
120  * bits set.
121  * e.g., _BITMASK(5) constructs '0x1F', _BITMASK(16) == 0xFFFF
122  * The symbol is intended to be used to limit the bit field width
123  * thusly:
124  * <a_register> = (any_expression) & _BITMASK(x), where 0 < x <= 32.
125  * If "any_expression" results in a value that is larger than can be
126  * contained in 'x' bits, the bits above 'x - 1' are masked off. When
127  * used with the _SBF example above, the example would be written:
128  * a_reg = ((_SBF(5,7) | _SBF(12,0xF)) & _BITMASK(16))
129  * This ensures that the value written to a_reg is no wider than
130  * 16 bits, and makes the code easier to read and understand.
131  */
132 #undef _BITMASK
133 /* Bitmask creation macro */
134 #define _BITMASK(field_width) ( _BIT(field_width) - 1)
