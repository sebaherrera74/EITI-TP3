84 /* Master transfer state change handler handler */
85 uint32_t Chip_I2CM_XferHandler(LPC_I2C_T *pI2C, I2CM_XFER_T *xfer)
86 {
87  uint32_t cclr = I2C_CON_FLAGS;
88 
89  switch (Chip_I2CM_GetCurState(pI2C)) {
90  case 0x08: /* Start condition on bus */
91  case 0x10: /* Repeated start condition */
92  pI2C->DAT = (xfer->slaveAddr << 1) | (xfer->txSz == 0);
93  break;
94 
95  /* Tx handling */
96  case 0x20: /* SLA+W sent NAK received */
97  case 0x30: /* DATA sent NAK received */
98  if ((xfer->options & I2CM_XFER_OPTION_IGNORE_NACK) == 0) {
99  xfer->status = I2CM_STATUS_NAK;
100  cclr &= ~I2C_CON_STO;
101  break;
102  }
103 
104  case 0x18: /* SLA+W sent and ACK received */
105  case 0x28: /* DATA sent and ACK received */
106  if (!xfer->txSz) {
107  if (xfer->rxSz) {
108  cclr &= ~I2C_CON_STA;
109  }
110  else {
111  xfer->status = I2CM_STATUS_OK;
112  cclr &= ~I2C_CON_STO;
113  }
114 
115  }
116  else {
117  pI2C->DAT = *xfer->txBuff++;
118  xfer->txSz--;
119  }
120  break;
121 
122  /* Rx handling */
123  case 0x58: /* Data Received and NACK sent */
124  case 0x50: /* Data Received and ACK sent */
125  *xfer->rxBuff++ = pI2C->DAT;
126  xfer->rxSz--;
127 
128  case 0x40: /* SLA+R sent and ACK received */
129  if ((xfer->rxSz > 1) || (xfer->options & I2CM_XFER_OPTION_LAST_RX_ACK)) {
130  cclr &= ~I2C_CON_AA;
131  }
132  if (xfer->rxSz == 0) {
133  xfer->status = I2CM_STATUS_OK;
134  cclr &= ~I2C_CON_STO;
135  }
136  break;
137 
138  /* NAK Handling */
139  case 0x48: /* SLA+R sent NAK received */
140  xfer->status = I2CM_STATUS_SLAVE_NAK;
141  cclr &= ~I2C_CON_STO;
142  break;
143 
144  case 0x38: /* Arbitration lost */
145  xfer->status = I2CM_STATUS_ARBLOST;
146  break;
147 
148  case 0x00: /* Bus Error */
149  xfer->status = I2CM_STATUS_BUS_ERROR;
150  cclr &= ~I2C_CON_STO;
151  break;
152  case 0xF8:
153  return 0;
154 
155  default:
156  xfer->status = I2CM_STATUS_ERROR;
157  cclr &= ~I2C_CON_STO;
158  break;
159  }
160 
161  /* Set clear control flags */
162  pI2C->CONSET = cclr ^ I2C_CON_FLAGS;
163  /* Stop flag should not be cleared as it is a reserved bit */
164  pI2C->CONCLR = cclr & (I2C_CON_AA | I2C_CON_SI | I2C_CON_STA);
165 
166  return xfer->status != I2CM_STATUS_BUSY;
167 }

170 void Chip_I2CM_Xfer(LPC_I2C_T *pI2C, I2CM_XFER_T *xfer)

180 /* Transmit and Receive data in master mode */
181 uint32_t Chip_I2CM_XferBlocking(LPC_I2C_T *pI2C, I2CM_XFER_T *xfer)
182 {
183  uint32_t ret = 0;
184  /* start transfer */
185  Chip_I2CM_Xfer(pI2C, xfer);
186 
187  while (ret == 0) {
188  /* wait for status change interrupt */
189  while ( Chip_I2CM_StateChanged(pI2C) == 0) {}
190  /* call state change handler */
191  ret = Chip_I2CM_XferHandler(pI2C, xfer);
192  }
193  return ret;
194 }

196 /* Master tx only */
197 uint32_t Chip_I2CM_Write(LPC_I2C_T *pI2C, const uint8_t *buff, uint32_t len)
198 {
199  uint32_t txLen = 0, err = 0;
200 
201  /* clear state change interrupt status */
202  Chip_I2CM_ClearSI(pI2C);
203  /* generate START condition */
204  Chip_I2CM_SendStart(pI2C);
205 
206  while ((txLen < len) && (err == 0)) {
207  /* wait for status change interrupt */
208  while ( Chip_I2CM_StateChanged(pI2C) == 0) {}
209 
210  /* check status and send data */
211  switch (Chip_I2CM_GetCurState(pI2C)) {
212  case 0x08: /* Start condition on bus */
213  case 0x10: /* Repeated start condition */
214  case 0x18: /* SLA+W sent and ACK received */
215  case 0x28: /* DATA sent and ACK received */
216  Chip_I2CM_WriteByte(pI2C, buff[txLen++]);
217  break;
218 
219  case 0x38: /* Arbitration lost */
220  break;
221 
222  default: /* we shouldn't be in any other state */
223  err = 1;
224  break;
225  }
226  /* clear state change interrupt status */
227  Chip_I2CM_ClearSI(pI2C);
228  }
229 
230  return txLen;
231 }

233 /* Sequential master read */
234 uint32_t Chip_I2CM_Read(LPC_I2C_T *pI2C, uint8_t *buff, uint32_t len)
235 {
236  uint32_t rxLen = 0, err = 0;
237 
238  /* clear state change interrupt status */
239  Chip_I2CM_ClearSI(pI2C);
240  /* generate START condition and auto-ack data received */
241  pI2C->CONSET = I2C_CON_AA | I2C_CON_STA;
242 
243  while ((rxLen < len) && (err == 0)) {
244  /* wait for status change interrupt */
245  while ( Chip_I2CM_StateChanged(pI2C) == 0) {}
246 
247  /* check status and send data */
248  switch (Chip_I2CM_GetCurState(pI2C)) {
249  case 0x08: /* Start condition on bus */
250  case 0x10: /* Repeated start condition */
251  case 0x40: /* SLA+R sent and ACK received */
252  case 0x50: /* Data Received and ACK sent */
253  buff[rxLen++] = Chip_I2CM_ReadByte(pI2C);
254  break;
255 
256  case 0x38: /* Arbitration lost */
257  break;
258 
259  default: /* we shouldn't be in any other state */
260  err = 1;
261  break;
262  }
263  /* clear state change interrupt status */
264  Chip_I2CM_ClearSI(pI2C);
265  }
266 
267  return rxLen;
268 }
