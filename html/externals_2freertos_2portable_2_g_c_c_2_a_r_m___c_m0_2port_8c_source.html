132 StackType_t * pxPortInitialiseStack( StackType_t * pxTopOfStack,
133  TaskFunction_t pxCode,
134  void * pvParameters )
135 {
136  /* Simulate the stack frame as it would be created by a context switch
137  * interrupt. */
138  pxTopOfStack--; /* Offset added to account for the way the MCU uses the stack on entry/exit of interrupts. */
139  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
140  pxTopOfStack--;
141  *pxTopOfStack = ( StackType_t ) pxCode; /* PC */
142  pxTopOfStack--;
143  *pxTopOfStack = ( StackType_t ) portTASK_RETURN_ADDRESS; /* LR */
144  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
145  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
146  pxTopOfStack -= 8; /* R11..R4. */
147 
148  return pxTopOfStack;
149 }

152 static void prvTaskExitError( void )
153 {
154  volatile uint32_t ulDummy = 0UL;
155 
156  /* A function that implements a task must not exit or attempt to return to
157  * its caller as there is nothing to return to. If a task wants to exit it
158  * should instead call vTaskDelete( NULL ).
159  *
160  * Artificially force an assert() to be triggered if configASSERT() is
161  * defined, then stop here so application writers can catch the error. */
162  configASSERT( uxCriticalNesting == ~0UL );
163  portDISABLE_INTERRUPTS();
164 
165  while( ulDummy == 0 )
166  {
167  /* This file calls prvTaskExitError() after the scheduler has been
168  * started to remove a compiler warning about the function being defined
169  * but never called. ulDummy is used purely to quieten other warnings
170  * about code appearing after this function is called - making ulDummy
171  * volatile makes the compiler think the function could return and
172  * therefore not output an 'unreachable code' warning for code that appears
173  * after it. */
174  }
175 }

185 void vPortStartFirstTask( void )
186 {
187  /* The MSP stack is not reset as, unlike on M3/4 parts, there is no vector
188  * table offset register that can be used to locate the initial stack value.
189  * Not all M0 parts have the application vector table at address 0. */
190  __asm volatile (
191  " .syntax unified \n"
192  " ldr r2, pxCurrentTCBConst2 \n"/* Obtain location of pxCurrentTCB. */
193  " ldr r3, [r2] \n"
194  " ldr r0, [r3] \n"/* The first item in pxCurrentTCB is the task top of stack. */
195  " adds r0, #32 \n"/* Discard everything up to r0. */
196  " msr psp, r0 \n"/* This is now the new top of stack to use in the task. */
197  " movs r0, #2 \n"/* Switch to the psp stack. */
198  " msr CONTROL, r0 \n"
199  " isb \n"
200  " pop {r0-r5} \n"/* Pop the registers that are saved automatically. */
201  " mov lr, r5 \n"/* lr is now in r5. */
202  " pop {r3} \n"/* Return address is now in r3. */
203  " pop {r2} \n"/* Pop and discard XPSR. */
204  " cpsie i \n"/* The first task has its context and interrupts can be enabled. */
205  " bx r3 \n"/* Finally, jump to the user defined task code. */
206  " \n"
207  " .align 4 \n"
208  "pxCurrentTCBConst2: .word pxCurrentTCB "
209  );
210 }

216 BaseType_t xPortStartScheduler( void )
217 {
218  /* Make PendSV, CallSV and SysTick the same priority as the kernel. */
219  portNVIC_SHPR3_REG |= portNVIC_PENDSV_PRI;
220  portNVIC_SHPR3_REG |= portNVIC_SYSTICK_PRI;
221 
222  /* Start the timer that generates the tick ISR. Interrupts are disabled
223  * here already. */
224  vPortSetupTimerInterrupt();
225 
226  /* Initialise the critical nesting count ready for the first task. */
227  uxCriticalNesting = 0;
228 
229  /* Start the first task. */
230  vPortStartFirstTask();
231 
232  /* Should never get here as the tasks will now be executing! Call the task
233  * exit error function to prevent compiler warnings about a static function
234  * not being called in the case that the application writer overrides this
235  * functionality by defining configTASK_RETURN_ADDRESS. Call
236  * vTaskSwitchContext() so link time optimisation does not remove the
237  * symbol. */
238  vTaskSwitchContext();
239  prvTaskExitError();
240 
241  /* Should not get here! */
242  return 0;
243 }

308 void xPortPendSVHandler( void )
309 {
310  /* This is a naked function. */
311 
312  __asm volatile
313  (
314  " .syntax unified \n"
315  " mrs r0, psp \n"
316  " \n"
317  " ldr r3, pxCurrentTCBConst \n"/* Get the location of the current TCB. */
318  " ldr r2, [r3] \n"
319  " \n"
320  " subs r0, r0, #32 \n"/* Make space for the remaining low registers. */
321  " str r0, [r2] \n"/* Save the new top of stack. */
322  " stmia r0!, {r4-r7} \n"/* Store the low registers that are not saved automatically. */
323  " mov r4, r8 \n"/* Store the high registers. */
324  " mov r5, r9 \n"
325  " mov r6, r10 \n"
326  " mov r7, r11 \n"
327  " stmia r0!, {r4-r7} \n"
328  " \n"
329  " push {r3, r14} \n"
330  " cpsid i \n"
331  " bl vTaskSwitchContext \n"
332  " cpsie i \n"
333  " pop {r2, r3} \n"/* lr goes in r3. r2 now holds tcb pointer. */
334  " \n"
335  " ldr r1, [r2] \n"
336  " ldr r0, [r1] \n"/* The first item in pxCurrentTCB is the task top of stack. */
337  " adds r0, r0, #16 \n"/* Move to the high registers. */
338  " ldmia r0!, {r4-r7} \n"/* Pop the high registers. */
339  " mov r8, r4 \n"
340  " mov r9, r5 \n"
341  " mov r10, r6 \n"
342  " mov r11, r7 \n"
343  " \n"
344  " msr psp, r0 \n"/* Remember the new top of stack for the task. */
345  " \n"
346  " subs r0, r0, #32 \n"/* Go back for the low registers that are not automatically restored. */
347  " ldmia r0!, {r4-r7} \n"/* Pop low registers. */
348  " \n"
349  " bx r3 \n"
350  " \n"
351  " .align 4 \n"
352  "pxCurrentTCBConst: .word pxCurrentTCB "
353  );
354 }

401  __attribute__( ( weak ) ) void vPortSuppressTicksAndSleep( TickType_t xExpectedIdleTime )
402  {
403  uint32_t ulReloadValue, ulCompleteTickPeriods, ulCompletedSysTickDecrements;
404  TickType_t xModifiableIdleTime;
405 
406  /* Make sure the SysTick reload value does not overflow the counter. */
407  if( xExpectedIdleTime > xMaximumPossibleSuppressedTicks )
408  {
409  xExpectedIdleTime = xMaximumPossibleSuppressedTicks;
410  }
411 
412  /* Stop the SysTick momentarily. The time the SysTick is stopped for
413  * is accounted for as best it can be, but using the tickless mode will
414  * inevitably result in some tiny drift of the time maintained by the
415  * kernel with respect to calendar time. */
416  portNVIC_SYSTICK_CTRL_REG &= ~portNVIC_SYSTICK_ENABLE_BIT;
417 
418  /* Calculate the reload value required to wait xExpectedIdleTime
419  * tick periods. -1 is used because this code will execute part way
420  * through one of the tick periods. */
421  ulReloadValue = portNVIC_SYSTICK_CURRENT_VALUE_REG + ( ulTimerCountsForOneTick * ( xExpectedIdleTime - 1UL ) );
422 
423  if( ulReloadValue > ulStoppedTimerCompensation )
424  {
425  ulReloadValue -= ulStoppedTimerCompensation;
426  }
427 
428  /* Enter a critical section but don't use the taskENTER_CRITICAL()
429  * method as that will mask interrupts that should exit sleep mode. */
430  __asm volatile ( "cpsid i" ::: "memory" );
431  __asm volatile ( "dsb" );
432  __asm volatile ( "isb" );
433 
434  /* If a context switch is pending or a task is waiting for the scheduler
435  * to be unsuspended then abandon the low power entry. */
436  if( eTaskConfirmSleepModeStatus() == eAbortSleep )
437  {
438  /* Restart from whatever is left in the count register to complete
439  * this tick period. */
440  portNVIC_SYSTICK_LOAD_REG = portNVIC_SYSTICK_CURRENT_VALUE_REG;
441 
442  /* Restart SysTick. */
443  portNVIC_SYSTICK_CTRL_REG |= portNVIC_SYSTICK_ENABLE_BIT;
444 
445  /* Reset the reload register to the value required for normal tick
446  * periods. */
447  portNVIC_SYSTICK_LOAD_REG = ulTimerCountsForOneTick - 1UL;
448 
449  /* Re-enable interrupts - see comments above the cpsid instruction()
450  * above. */
451  __asm volatile ( "cpsie i" ::: "memory" );
452  }
453  else
454  {
455  /* Set the new reload value. */
456  portNVIC_SYSTICK_LOAD_REG = ulReloadValue;
457 
458  /* Clear the SysTick count flag and set the count value back to
459  * zero. */
460  portNVIC_SYSTICK_CURRENT_VALUE_REG = 0UL;
461 
462  /* Restart SysTick. */
463  portNVIC_SYSTICK_CTRL_REG |= portNVIC_SYSTICK_ENABLE_BIT;
464 
465  /* Sleep until something happens. configPRE_SLEEP_PROCESSING() can
466  * set its parameter to 0 to indicate that its implementation contains
467  * its own wait for interrupt or wait for event instruction, and so wfi
468  * should not be executed again. However, the original expected idle
469  * time variable must remain unmodified, so a copy is taken. */
470  xModifiableIdleTime = xExpectedIdleTime;
471  configPRE_SLEEP_PROCESSING( xModifiableIdleTime );
472 
473  if( xModifiableIdleTime > 0 )
474  {
475  __asm volatile ( "dsb" ::: "memory" );
476  __asm volatile ( "wfi" );
477  __asm volatile ( "isb" );
478  }
479 
480  configPOST_SLEEP_PROCESSING( xExpectedIdleTime );
481 
482  /* Re-enable interrupts to allow the interrupt that brought the MCU
483  * out of sleep mode to execute immediately. see comments above
484  * __disable_interrupt() call above. */
485  __asm volatile ( "cpsie i" ::: "memory" );
486  __asm volatile ( "dsb" );
487  __asm volatile ( "isb" );
488 
489  /* Disable interrupts again because the clock is about to be stopped
490  * and interrupts that execute while the clock is stopped will increase
491  * any slippage between the time maintained by the RTOS and calendar
492  * time. */
493  __asm volatile ( "cpsid i" ::: "memory" );
494  __asm volatile ( "dsb" );
495  __asm volatile ( "isb" );
496 
497  /* Disable the SysTick clock without reading the
498  * portNVIC_SYSTICK_CTRL_REG register to ensure the
499  * portNVIC_SYSTICK_COUNT_FLAG_BIT is not cleared if it is set. Again,
500  * the time the SysTick is stopped for is accounted for as best it can
501  * be, but using the tickless mode will inevitably result in some tiny
502  * drift of the time maintained by the kernel with respect to calendar
503  * time*/
504  portNVIC_SYSTICK_CTRL_REG = ( portNVIC_SYSTICK_CLK_BIT | portNVIC_SYSTICK_INT_BIT );
505 
506  /* Determine if the SysTick clock has already counted to zero and
507  * been set back to the current reload value (the reload back being
508  * correct for the entire expected idle time) or if the SysTick is yet
509  * to count to zero (in which case an interrupt other than the SysTick
510  * must have brought the system out of sleep mode). */
511  if( ( portNVIC_SYSTICK_CTRL_REG & portNVIC_SYSTICK_COUNT_FLAG_BIT ) != 0 )
512  {
513  uint32_t ulCalculatedLoadValue;
514 
515  /* The tick interrupt is already pending, and the SysTick count
516  * reloaded with ulReloadValue. Reset the
517  * portNVIC_SYSTICK_LOAD_REG with whatever remains of this tick
518  * period. */
519  ulCalculatedLoadValue = ( ulTimerCountsForOneTick - 1UL ) - ( ulReloadValue - portNVIC_SYSTICK_CURRENT_VALUE_REG );
520 
521  /* Don't allow a tiny value, or values that have somehow
522  * underflowed because the post sleep hook did something
523  * that took too long. */
524  if( ( ulCalculatedLoadValue < ulStoppedTimerCompensation ) || ( ulCalculatedLoadValue > ulTimerCountsForOneTick ) )
525  {
526  ulCalculatedLoadValue = ( ulTimerCountsForOneTick - 1UL );
527  }
528 
529  portNVIC_SYSTICK_LOAD_REG = ulCalculatedLoadValue;
530 
531  /* As the pending tick will be processed as soon as this
532  * function exits, the tick value maintained by the tick is stepped
533  * forward by one less than the time spent waiting. */
534  ulCompleteTickPeriods = xExpectedIdleTime - 1UL;
535  }
536  else
537  {
538  /* Something other than the tick interrupt ended the sleep.
539  * Work out how long the sleep lasted rounded to complete tick
540  * periods (not the ulReload value which accounted for part
541  * ticks). */
542  ulCompletedSysTickDecrements = ( xExpectedIdleTime * ulTimerCountsForOneTick ) - portNVIC_SYSTICK_CURRENT_VALUE_REG;
543 
544  /* How many complete tick periods passed while the processor
545  * was waiting? */
546  ulCompleteTickPeriods = ulCompletedSysTickDecrements / ulTimerCountsForOneTick;
547 
548  /* The reload value is set to whatever fraction of a single tick
549  * period remains. */
550  portNVIC_SYSTICK_LOAD_REG = ( ( ulCompleteTickPeriods + 1UL ) * ulTimerCountsForOneTick ) - ulCompletedSysTickDecrements;
551  }
552 
553  /* Restart SysTick so it runs from portNVIC_SYSTICK_LOAD_REG
554  * again, then set portNVIC_SYSTICK_LOAD_REG back to its standard
555  * value. */
556  portNVIC_SYSTICK_CURRENT_VALUE_REG = 0UL;
557  portNVIC_SYSTICK_CTRL_REG |= portNVIC_SYSTICK_ENABLE_BIT;
558  vTaskStepTick( ulCompleteTickPeriods );
559  portNVIC_SYSTICK_LOAD_REG = ulTimerCountsForOneTick - 1UL;
560 
561  /* Exit with interrpts enabled. */
562  __asm volatile ( "cpsie i" ::: "memory" );
563  }
564  }
