87 static void pll_calc_divs(uint32_t freq, PLL_PARAM_T *ppll)
88 {
89 
90  uint32_t prev = freq;
91  int n, m, p;
92 
93  /* When direct mode is set FBSEL should be a don't care */
94  if (ppll->ctrl & (1 << 7)) {
95  ppll->ctrl &= ~(1 << 6);
96  }
97  for (n = 1; n <= 4; n++) {
98  for (p = 0; p < 4; p ++) {
99  for (m = 1; m <= 256; m++) {
100  uint32_t fcco, fout;
101  if (ppll->ctrl & (1 << 6)) {
102  fcco = ((m << (p + 1)) * ppll->fin) / n;
103  } else {
104  fcco = (m * ppll->fin) / n;
105  }
106  if (fcco < PLL_MIN_CCO_FREQ) continue;
107  if (fcco > PLL_MAX_CCO_FREQ) break;
108  if (ppll->ctrl & (1 << 7)) {
109  fout = fcco;
110  } else {
111  fout = fcco >> (p + 1);
112  }
113 
114  if (ABS(freq - fout) < prev) {
115  ppll->nsel = n;
116  ppll->psel = p + 1;
117  ppll->msel = m;
118  ppll->fout = fout;
119  ppll->fcco = fcco;
120  prev = ABS(freq - fout);
121  }
122  }
123  }
124  }
125 }

127 static void pll_get_frac(uint32_t freq, PLL_PARAM_T *ppll)
128 {
129  int diff[3];
130  PLL_PARAM_T pll[3] = {{0},{0},{0}};
131 
132  /* Try direct mode */
133  pll[0].ctrl |= (1 << 7);
134  pll[0].fin = ppll->fin;
135  pll[0].srcin = ppll->srcin;
136  pll_calc_divs(freq, &pll[0]);
137  if (pll[0].fout == freq) {
138  *ppll = pll[0];
139  return ;
140  }
141  diff[0] = ABS(freq - pll[0].fout);
142 
143  /* Try non-Integer mode */
144  pll[2].ctrl &= ~(1 << 6); // need to set FBSEL to 0
145  pll[2].fin = ppll->fin;
146  pll[2].srcin = ppll->srcin;
147  pll_calc_divs(freq, &pll[2]);
148  if (pll[2].fout == freq) {
149  *ppll = pll[2];
150  return ;
151  }
152  diff[2] = ABS(freq - pll[2].fout);
153 
154  if (freq <= 110000000) {
155  /* Try integer mode */
156  pll[1].ctrl = (1 << 6);
157  pll[1].fin = ppll->fin;
158  pll_calc_divs(freq, &pll[1]);
159  if (pll[1].fout == freq) {
160  *ppll = pll[1];
161  return ;
162  }
163  }
164  diff[1] = ABS(freq - pll[1].fout);
165 
166  /* Find the min of 3 and return */
167  if (diff[0] <= diff[1]) {
168  if (diff[0] <= diff[2]) {
169  *ppll = pll[0];
170  } else {
171  *ppll = pll[2];
172  }
173  } else {
174  if (diff[1] <= diff[2]) {
175  *ppll = pll[1];
176  } else {
177  *ppll = pll[2];
178  }
179  }
180 }

252 /* Calculate the Main PLL div values */
253 int Chip_Clock_CalcMainPLLValue(uint32_t freq, PLL_PARAM_T *ppll)
254 {
255  ppll->fin = Chip_Clock_GetClockInputHz(ppll->srcin);
256 
257  /* Do sanity check on frequency */
258  if (freq > MAX_CLOCK_FREQ || freq < (PLL_MIN_CCO_FREQ / 16) || !ppll->fin) {
259  return -1;
260  }
261 
262  ppll->ctrl = 1 << 7; /* Enable direct mode [If possible] */
263  ppll->nsel = 0;
264  ppll->psel = 0;
265  ppll->msel = freq / ppll->fin;
266 
267  if (freq < PLL_MIN_CCO_FREQ || ppll->msel * ppll->fin != freq) {
268  pll_get_frac(freq, ppll);
269  if (!ppll->nsel) {
270  return -1;
271  }
272  ppll->nsel --;
273  }
274 
275  if (ppll->msel == 0) {
276  return - 1;
277  }
278 
279  if (ppll->psel) {
280  ppll->psel --;
281  }
282 
283  ppll->msel --;
284 
285  return 0;
286 }

295 /* Configures the main PLL */
296 uint32_t Chip_Clock_SetupMainPLLHz(CHIP_CGU_CLKIN_T Input, uint32_t MinHz, uint32_t DesiredHz, uint32_t MaxHz)
297 {
298  uint32_t freqin = Chip_Clock_GetClockInputHz(Input);
299  uint32_t Mult, LastMult, MultEnd;
300  uint32_t freqout, freqout2;
301 
302  if (DesiredHz != 0xFFFFFFFF) {
303  /* Test DesiredHz rounded down */
304  Mult = DesiredHz / freqin;
305  freqout = Chip_Clock_TestMainPLLMultiplier(freqin, Mult, MinHz, MaxHz);
306 
307  /* Test DesiredHz rounded up */
308  Mult++;
309  freqout2 = Chip_Clock_TestMainPLLMultiplier(freqin, Mult, MinHz, MaxHz);
310 
311  if (freqout && !freqout2) { /* rounding up is no good? set first multiplier */
312  Mult--;
313  return Chip_Clock_SetupMainPLLMult(Input, Mult);
314  }
315  if (!freqout && freqout2) { /* didn't work until rounded up? set 2nd multiplier */
316  return Chip_Clock_SetupMainPLLMult(Input, Mult);
317  }
318 
319  if (freqout && freqout2) { /* either multiplier okay? choose closer one */
320  if ((DesiredHz - freqout) > (freqout2 - DesiredHz)) {
321  Mult--;
322  return Chip_Clock_SetupMainPLLMult(Input, Mult);
323  }
324  else {
325  return Chip_Clock_SetupMainPLLMult(Input, Mult);
326  }
327  }
328  }
329 
330  /* Neither multiplier okay? Try to start at MinHz and increment.
331  This should find the highest multiplier that is still good */
332  Mult = MinHz / freqin;
333  MultEnd = MaxHz / freqin;
334  LastMult = 0;
335  while (1) {
336  freqout = Chip_Clock_TestMainPLLMultiplier(freqin, Mult, MinHz, MaxHz);
337 
338  if (freqout) {
339  LastMult = Mult;
340  }
341 
342  if (Mult >= MultEnd) {
343  break;
344  }
345  Mult++;
346  }
347 
348  if (LastMult) {
349  return Chip_Clock_SetupMainPLLMult(Input, LastMult);
350  }
351 
352  return 0;
353 }

355 /* Directly set the PLL multipler */
356 uint32_t Chip_Clock_SetupMainPLLMult(CHIP_CGU_CLKIN_T Input, uint32_t mult)
357 {
358  volatile uint32_t delay = 250;
359  uint32_t freq = Chip_Clock_GetClockInputHz(Input);
360  uint32_t msel = 0, nsel = 0, psel = 0, pval = 1;
361  uint32_t PLLReg = LPC_CGU->PLL1_CTRL;
362 
363  freq *= mult;
364  msel = mult - 1;
365 
366  PLLReg &= ~(0x1F << 24);/* clear input source bits */
367  PLLReg |= Input << 24; /* set input source bits to parameter */
368 
369  /* Clear other PLL input bits */
370  PLLReg &= ~((1 << 6) | /* FBSEL */
371  (1 << 1) | /* BYPASS */
372  (1 << 7) | /* DIRECT */
373  (0x03 << 8) | (0xFF << 16) | (0x03 << 12)); /* PSEL, MSEL, NSEL- divider ratios */
374 
375  PLLReg |= (1 << 11); /* AUTOBLOCK */
376 
377  if (freq < 156000000) {
378  /* psel is encoded such that 0=1, 1=2, 2=4, 3=8 */
379  while ((2 * (pval) * freq) < 156000000) {
380  psel++;
381  pval *= 2;
382  }
383 
384  PLLReg |= (msel << 16) | (nsel << 12) | (psel << 8) | (1 << 6); /* dividers + FBSEL */
385  }
386  else if (freq < 320000000) {
387  PLLReg |= (msel << 16) | (nsel << 12) | (psel << 8) | (1 << 7) | (1 << 6); /* dividers + DIRECT + FBSEL */
388  }
389  else {
390  Chip_Clock_DisableMainPLL();
391  return 0;
392  }
393  LPC_CGU->PLL1_CTRL = PLLReg & ~(1 << 0);
394 
395  /* Wait for 50uSec */
396  while(delay--) {}
397 
398  return freq;
399 }

401 /* Returns the frequency of the main PLL */
402 uint32_t Chip_Clock_GetMainPLLHz(void)
403 {
404  uint32_t PLLReg = LPC_CGU->PLL1_CTRL;
405  uint32_t freq = Chip_Clock_GetClockInputHz((CHIP_CGU_CLKIN_T) ((PLLReg >> 24) & 0xF));
406  uint32_t msel, nsel, psel, direct, fbsel;
407  uint32_t m, n, p;
408  const uint8_t ptab[] = {1, 2, 4, 8};
409 
410  /* No lock? */
411  if (!(LPC_CGU->PLL1_STAT & 1)) {
412  return 0;
413  }
414 
415  msel = (PLLReg >> 16) & 0xFF;
416  nsel = (PLLReg >> 12) & 0x3;
417  psel = (PLLReg >> 8) & 0x3;
418  direct = (PLLReg >> 7) & 0x1;
419  fbsel = (PLLReg >> 6) & 0x1;
420 
421  m = msel + 1;
422  n = nsel + 1;
423  p = ptab[psel];
424 
425  if (direct || fbsel) {
426  return m * (freq / n);
427  }
428 
429  return (m / (2 * p)) * (freq / n);
430 }

433 void Chip_Clock_SetDivider(CHIP_CGU_IDIV_T Divider, CHIP_CGU_CLKIN_T Input, uint32_t Divisor)
434 {
435  uint32_t reg = LPC_CGU->IDIV_CTRL[Divider];
436 
437  Divisor--;
438 
439  if (Input != CLKINPUT_PD) {
440  /* Mask off bits that need to changes */

447  LPC_CGU->IDIV_CTRL[Divider] = reg | 1; /* Power down this divider */
448  }

734 /* Returns a peripheral clock rate */
735 uint32_t Chip_Clock_GetRate(CHIP_CCU_CLK_T clk)
736 {
737  CHIP_CGU_BASE_CLK_T baseclk;
738  uint32_t reg, div, rate;
739 
740  /* Get CCU config register for clock */
741  if (clk >= CLK_CCU2_START) {
742  reg = LPC_CCU2->CLKCCU[clk - CLK_CCU2_START].CFG;
743  }
744  else {
745  reg = LPC_CCU1->CLKCCU[clk].CFG;
746  }
747 
748  /* Is the clock enabled? */
749  if (reg & 1) {
750  /* Get base clock for this peripheral clock */
751  baseclk = Chip_Clock_FindBaseClock(clk);
752 
753  /* Get base clock rate */
754  rate = Chip_Clock_GetBaseClocktHz(baseclk);
755 
756  /* Get divider for this clock */
757  if (((reg >> 5) & 0x7) == 0) {
758  div = 1;
759  }
760  else {
761  div = 2;/* No other dividers supported */
762 
763  }
764  rate = rate / div;
765  }
766  else {
767  rate = 0;
768  }
769 
770  return rate;
771 }
